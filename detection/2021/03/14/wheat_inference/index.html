class WheatDataLoader(Dataset): # 옵션들이 반복해서 사용되므로 함수 및 변수를 모듈화한 Dataset을 상속 받아서 사용한다.
    def __init__(self, dataframe, image_dir, transforms=None):
        super().__init__()
        self.df = dataframe
        self.image_dir = image_dir
        self.transforms = transforms
        self.image_ids = dataframe['image_id'].unique() # unique한 3373개의 이미지 각각에 대한 id를 가져온다.
        
    def __getitem__(self, index): 
        
        # Dataset의 index와 연동되어 이미지 각각에 대해 접근한다.
        image_id = self.image_ids[index] 
        
        # cv2.IMREAD_COLOR 옵션은 이미지를 3개의 채널로 가져오겠다는 의미이다.
        image = cv2.imread(self.image_dir + image_id + '.jpg', cv2.IMREAD_COLOR)
        
        # cv2로 이미지를 가져오면 (R,G,B)가 아닌 (B,G,R) 순서로 가져온다. 따라서 cv2.COLOR_BGR2RGB 옵션으로 순서를 바꿔주자.
        # 또한 메모리 소모량을 줄이기 위해 데이터타입을 float32로 바꾼다.
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB).astype(np.float32) 
        
        # keras에서는 이미지의 픽셀 값들을 0~1 사이로 자동적으로 표준화를 해주지만, 여기선 직접 설정해줘야 한다.
        image = image / 255.0
        
        # transforms에서 옵션을 넣어주어 array형식을 tensor로 바꿔줘야 한다. 
        # 모델 안에서 gradient값을 축적, 저장할 수 있게끔 데이터를 담는 그릇만 학습 가능한 형태로 바꿔주는 것이다.
        if self.transforms :
            sample = {'image' : image}
            
            # ** : transforms 함수 안에 'image'라는 이름에 접근한다는 의미
            sample = self.transforms(**sample)
            
            image = sample['image']
        
        return image, image_id
        
    def __len__(self):
        return len(self.image_ids)

모델 가중치 로드하기

  • kaggle에서는 쉽게 fastrcnn의 가중치 파일을 찾을 수 있다.
  • +Add data -> fasterrcnn_resnet50_fpn 검색 -> weights add 하기 -> input/weights에서 모델 경로 가져오기
  • 추가하고자 하는 모델 가중치가 gpu 환경에서 학습되었기 때문에, 우리의 모델 역시 gpu 환경으로 맞춰준다.
  • 또한 데이터를 예측하고 싶을 때는 반드시 model.eval()옵션을 준다. 이 옵션을 사용하면 모델 내부의 모든 layer가 evaluation 모드가 된다. Evaluation 모드에서는 batchnorm, Dropout과 같은 기능들이 사용되지 않는다.

def format_string(boxes, score):
    pred_strings = []
    for j in zip(score, boxes):
        
        pred_strings.append("{0:.4f} {1} {2} {3} {4}".format(j[0], j[1][0], j[1][1], j[1][2], j[1][3]))
        
    return " ".join(pred_strings)

results = []
sample_bb = []
sample_id = []
idx = 0

for images, image_ids in test_loader:
    
    # 데이터 역시 gpu 환경으로 바꿔준다
    images = list(x.to(args["DEVICE"]) for x in images)

    # predict -> class, boundingbox
    outputs = model(images)
    
    # 제출 양식에 맞게 결과 처리하기
    for i, image in enumerate(images):
        boxes = outputs[i]['boxes'].data.cpu().numpy()
        scores = outputs[i]['scores'].data.cpu().numpy()
        # 예측된 모든 bounding box를 가져오기 때문에, 일정 확률 이상의 box들만 가져온다
        
        boxes = boxes[scores > 0.5]
        scores = scores[scores > 0.5]
        
        image_id = image_ids[i]
        
        # 결과 시각화를 위해 저장
        sample_id.append(image_id)
        sample_bb.append(boxes.copy()) # xmin, ymin, xmax, ymax

        # 좌표 전처리
        # xmax -> w
        boxes[:,2] = boxes[:,2] - boxes[:,0]
        # ymax -> h
        boxes[:,3] = boxes[:,3] - boxes[:,1]
                
        # 결과 저장
        result = {'image_id' : image_id, 'PredictionString' : format_string(boxes, scores)}
        
        results.append(result)
        

        if idx ==0:
            print(f'{idx+1} image predicted and result saved ...')
        else : 
            print(f'{idx+1} images predicted and result saved ...')
        idx+=1
